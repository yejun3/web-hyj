switch문 (조건문)
switch (조건) { 
case 값1 : 명령문
 break ; 
case 값2 : 명령문
break;
default : 명령문
// 해당 조건문은 동등 연산자를 사용
// break와 default는 생략 가능

예제 switch문
int num =2;
switch (num) { 
case 1  : 명령문
 break ; 
case 2 : 명령문 // case 2의 명령문이 출력
break;
default : 명령문

//특징1: 동등 연산자만 사용
//특징 2: 사용하기로한 변수(num)를 기준으로  case를 비교한다.(같은지 비교)
// 특징 3: case의 모든 내용이 실행되지 않을 때 default 사용
// 특징 4 : break라는 제어문 사용 
//break문은 switch에서 다음으로 넘어가지 않게 만드는 역할
